// autocompleteHandler returns quick suggestions (prefix + fuzzy)
func (s *SearchService) autocompleteHandler(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	q := strings.TrimSpace(r.URL.Query().Get("q"))
	limit := 10
	if q == "" {
		http.Error(w, "query param q is required", http.StatusBadRequest)
		return
	}

	// lowercase for consistent caching
	cacheKey := fmt.Sprintf("autocomplete:%s", strings.ToLower(q))
	if cached, err := s.RDB.Get(ctx, cacheKey).Result(); err == nil {
		w.Header().Set("Content-Type", "application/json")
		w.Write([]byte(cached))
		return
	}

	// Step 1: prefix matches (fast and prioritized)
	prefixResults, err := s.prefixSuggestions(ctx, q, limit)
	if err != nil {
		http.Error(w, fmt.Sprintf("prefix query error: %v", err), http.StatusInternalServerError)
		return
	}

	// If we have enough prefix results, return them
	if len(prefixResults) >= limit {
		s.cacheAndWrite(ctx, cacheKey, prefixResults)
		return
	}

	// Step 2: fuzzy matches using pg_trgm similarity (exclude already included IDs)
	remaining := limit - len(prefixResults)
	excludedIDs := make([]string, 0, len(prefixResults))
	for _, u := range prefixResults {
		excludedIDs = append(excludedIDs, u.ID)
	}
	fuzzyResults, err := s.fuzzySuggestions(ctx, q, remaining, excludedIDs)
	if err != nil {
		http.Error(w, fmt.Sprintf("fuzzy query error: %v", err), http.StatusInternalServerError)
		return
	}

	combined := append(prefixResults, fuzzyResults...)
	// Final small tie-break sort: popularity desc
	sort.SliceStable(combined, func(i, j int) bool {
		return combined[i].Followers > combined[j].Followers
	})

	s.cacheAndWrite(ctx, cacheKey, combined)
}

// getCandidates collects prefix and fuzzy candidates, merges them, returns up to limit
func (s *SearchService) getCandidates(ctx context.Context, q string, limit int, league, city string) ([]UserResult, error) {
	// 1) prefix candidates
	prefix, err := s.prefixSuggestions(ctx, q, limit)
	if err != nil {
		return nil, err
	}
	if len(prefix) >= limit {
		return prefix[:limit], nil
	}

	// 2) fuzzy candidates (exclude prefix IDs)
	excluded := make([]string, 0, len(prefix))
	for _, u := range prefix {
		excluded = append(excluded, u.ID)
	}
	toFetch := limit - len(prefix)
	fuzzy, err := s.fuzzySuggestions(ctx, q, toFetch, excluded)
	if err != nil {
		return nil, err
	}

	all := append(prefix, fuzzy...)
	// dedupe by ID (simple map)
	seen := map[string]bool{}
	final := make([]UserResult, 0, len(all))
	for _, u := range all {
		if !seen[u.ID] {
			final = append(final, u)
			seen[u.ID] = true
		}
	}
	return final, nil
}

// autocompleteHandler returns quick suggestions (prefix + fuzzy)
func (s *SearchService) autocompleteHandler(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	q := strings.TrimSpace(r.URL.Query().Get("q"))
	limit := 10
	if q == "" {
		http.Error(w, "query param q is required", http.StatusBadRequest)
		return
	}

	// lowercase for consistent caching
	cacheKey := fmt.Sprintf("autocomplete:%s", strings.ToLower(q))
	if cached, err := s.RDB.Get(ctx, cacheKey).Result(); err == nil {
		w.Header().Set("Content-Type", "application/json")
		w.Write([]byte(cached))
		return
	}

	// Step 1: prefix matches (fast and prioritized)
	prefixResults, err := s.prefixSuggestions(ctx, q, limit)
	if err != nil {
		http.Error(w, fmt.Sprintf("prefix query error: %v", err), http.StatusInternalServerError)
		return
	}

	// If we have enough prefix results, return them
	if len(prefixResults) >= limit {
		s.cacheAndWrite(ctx, cacheKey, prefixResults)
		return
	}

	// Step 2: fuzzy matches using pg_trgm similarity (exclude already included IDs)
	remaining := limit - len(prefixResults)
	excludedIDs := make([]string, 0, len(prefixResults))
	for _, u := range prefixResults {
		excludedIDs = append(excludedIDs, u.ID)
	}
	fuzzyResults, err := s.fuzzySuggestions(ctx, q, remaining, excludedIDs)
	if err != nil {
		http.Error(w, fmt.Sprintf("fuzzy query error: %v", err), http.StatusInternalServerError)
		return
	}

	combined := append(prefixResults, fuzzyResults...)
	// Final small tie-break sort: popularity desc
	sort.SliceStable(combined, func(i, j int) bool {
		return combined[i].Followers > combined[j].Followers
	})

	s.cacheAndWrite(ctx, cacheKey, combined)
}

// -------------------- AUTOCOMPLETE --------------------

func (s *SearchService) autocompleteHandler(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	q := strings.TrimSpace(r.URL.Query().Get("q"))
	if q == "" {
		http.Error(w, "query param q required", http.StatusBadRequest)
		return
	}
	limit := 10
	cacheKey := "auto:" + strings.ToLower(q)

	if cached, err := s.RDB.Get(ctx, cacheKey).Result(); err == nil {
		w.Header().Set("Content-Type", "application/json")
		w.Write([]byte(cached))
		return
	}

	prefix, _ := s.prefixSuggestions(ctx, q, limit)
	if len(prefix) >= limit {
		s.cacheWrite(ctx, cacheKey, prefix, 30*time.Second)
		writeJSON(w, prefix)
		return
	}

	excluded := make([]string, len(prefix))
	for i, p := range prefix {
		excluded[i] = p.ID
	}
	fuzzy, _ := s.fuzzySuggestions(ctx, q, limit-len(prefix), excluded)

	results := append(prefix, fuzzy...)
	sort.SliceStable(results, func(i, j int) bool {
		return results[i].Followers > results[j].Followers
	})

	s.cacheWrite(ctx, cacheKey, results, 30*time.Second)
	writeJSON(w, results)
}

func (s *SearchService) searchHandler(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	q := strings.TrimSpace(r.URL.Query().Get("q"))
	if q == "" {
		http.Error(w, "q required", http.StatusBadRequest)
		return
	}

	candidates, err := s.getCandidates(ctx, q, 50)
	if err != nil {
		http.Error(w, err.Error(), 500)
		return
	}
	if len(candidates) == 0 {
		writeJSON(w, []UserResult{})
		return
	}

	enriched := s.enrichCandidates(ctx, "", q, candidates)
	for i := range enriched {
		enriched[i].Score = computeFinalScore(enriched[i])
	}
	sort.SliceStable(enriched, func(i, j int) bool { return enriched[i].Score > enriched[j].Score })

	if len(enriched) > 20 {
		enriched = enriched[:20]
	}
	writeJSON(w, enriched)
}

func (s *SearchService) prefixSuggestions(ctx context.Context, q string, limit int) ([]UserResult, error) {
	sql := `
	SELECT id, username, full_name, followers, 1.0 AS text_score
	FROM users
	WHERE LOWER(username) LIKE LOWER($1) || '%' OR LOWER(full_name) LIKE LOWER($1) || '%'
	ORDER BY followers DESC
	LIMIT $2;`
	rows, err := s.DB.QueryContext(ctx, sql, q, limit)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var results []UserResult
	for rows.Next() {
		var u UserResult
		rows.Scan(&u.ID, &u.Username, &u.FullName, &u.Followers, &u.TextScore)
		results = append(results, u)
	}
	return results, nil
}
