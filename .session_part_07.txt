	rows, err := s.DB.QueryContext(ctx, sql, args...)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var results []UserResult
	for rows.Next() {
		var u UserResult
		rows.Scan(&u.ID, &u.Username, &u.FullName, &u.Followers, &u.TextScore)
		results = append(results, u)
	}
	return results, nil
}

func (s *SearchService) getCandidates(ctx context.Context, q string, limit int) ([]UserResult, error) {
	prefix, _ := s.prefixSuggestions(ctx, q, limit)
	excluded := make([]string, len(prefix))
	for i, p := range prefix {
		excluded[i] = p.ID
	}
	fuzzy, _ := s.fuzzySuggestions(ctx, q, limit-len(prefix), excluded)
	return append(prefix, fuzzy...), nil
}

func computeFinalScore(u enriched) float64 {
	wText, wPop, wInter, wMutual, wAct, wLoc, wAI := 0.3, 0.25, 0.15, 0.1, 0.1, 0.05, 0.05
	return wText*u.TextScore +
		wPop*math.Log1p(float64(u.Followers)) +
		wInter*u.InteractionScore +
		wMutual*float64(u.Mutuals) +
		wAct*u.ActivityScore +
		wLoc*u.LocationScore +
		wAI*u.EmbeddingScore
}

	TextRelevanceWeight   = 0.30
	PopularityWeight      = 0.20
	InteractionWeight     = 0.20
	SocialProximityWeight = 0.10
	LocationWeight        = 0.10
	ActivityWeight        = 0.10
)

// --- Core Algorithm ---

func rankUsers(users []User, query string, currentUserLocation string) []RankedUser {
	var ranked []RankedUser

	for _, user := range users {
		score := calculateWeightedScore(user, query, currentUserLocation)
		ranked = append(ranked, RankedUser{User: user, Score: score})
	}

	// Sort by score descending
	for i := 0; i < len(ranked)-1; i++ {
		for j := i + 1; j < len(ranked); j++ {
			if ranked[j].Score > ranked[i].Score {
				ranked[i], ranked[j] = ranked[j], ranked[i]
			}
		}
	}

	return ranked
}

// Weighted scoring
func calculateWeightedScore(user User, query string, currentUserLocation string) float64 {
	textScore := textRelevance(user, query)
	popularityScore := popularity(user)
	interactionScore := interaction(user)
	socialScore := socialProximity(user)
	locationScore := locationRelevance(user, currentUserLocation)
	activityScore := activity(user)

	total := (textScore * TextRelevanceWeight) +
		(popularityScore * PopularityWeight) +
		(interactionScore * InteractionWeight) +
		(socialScore * SocialProximityWeight) +
		(locationScore * LocationWeight) +
		(activityScore * ActivityWeight)

	return math.Round(total*100) / 100 // keep 2 decimals
}

// --- Individual Scoring Functions ---

func textRelevance(u User, query string) float64 {
	query = strings.ToLower(query)
	nameMatch := strings.Contains(strings.ToLower(u.FullName), query)
	userMatch := strings.Contains(strings.ToLower(u.Username), query)
	bioMatch := strings.Contains(strings.ToLower(u.Bio), query)

	matchCount := 0
	if nameMatch {
		matchCount++
	}
	if userMatch {
		matchCount++
	}
	if bioMatch {
		matchCount++
	}

	return float64(matchCount) / 3.0 // normalize 0–1
}

func popularity(u User) float64 {
	total := float64(u.Followers + u.ChallengesWon*2 - u.ChallengesLost)
	return normalize(total, 0, 1000) // assume 1000 is upper limit baseline
}

func interaction(u User) float64 {
	total := float64(u.LikesReceived + u.DMCount)
	return normalize(total, 0, 500)
}

func socialProximity(u User) float64 {
	return normalize(float64(u.MutualCount), 0, 50)
}

func locationRelevance(u User, current string) float64 {
	if strings.EqualFold(u.Location, current) {
		return 1.0
	}
	return 0.0
}

func activity(u User) float64 {
	hoursAgo := time.Since(u.LastActiveAt).Hours()
	if hoursAgo < 1 {
		return 1.0
	} else if hoursAgo < 24 {
		return 0.8
	} else if hoursAgo < 72 {
		return 0.5
	}
	return 0.2
}

// Helper normalization
func normalize(value, min, max float64) float64 {
	if value <= min {
		return 0
	}
	if value >= max {
		return 1
	}
	return (value - min) / (max - min)
}

**Better approach:**

```
API Gateway (single entry point)
    ├── Auth Service (login, signup, tokens)
    ├── User Service (profiles, follows, settings)
    ├── Feed Service (ranking, serving shorts)
    ├── Challenge Service (create, accept, judge, result)
    ├── Notification Service (WebSocket + FCM + Redis)
    ├── Search Service (ranking, autocomplete)
    ├── Media Service (upload, transcode, CDN)
    └── Analytics Service (watch time, interactions, mood inference)
```

**You don't need to build all microservices now**, but structure your code in separate packages so splitting later is painless.

**What you're missing:**
- **No API Gateway** — you need rate limiting, auth middleware, request validation at a single entry point
- **No message queue** — your notification system calls functions directly; when one fails, everything fails. Use Redis Streams or even a simple Go channel-based queue
- **No health checks** — if your WebSocket service crashes, nothing restarts it
- **No graceful shutdown** — your server doesn't handle `SIGTERM`, meaning active connections get killed during deploys

## 2. AUTHENTICATION — Fundamentally Broken

**Problem:** Your current auth is username+password with no tokens. The Flutter app has no session persistence. Every app restart requires re-login.

**What you need:**

- **JWT tokens** — access token (15 min) + refresh token (30 days)
- **Token rotation** — refresh token is single-use; issuing a new one invalidates the old
- **Device fingerprinting** — store device ID with refresh token so stolen tokens can't be used on different devices
- **Rate limiting on login** — max 5 attempts per 15 minutes per IP
- **Password hashing** — bcrypt with cost ≥ 12 (verify your current implementation)
- **OAuth later** — Google/Apple Sign-In removes password friction entirely

**Your current login returns ALL users to the client.** This is a massive security and performance problem. A user with 1 million other users in the system would download all of them on login. Remove this entirely — fetch users on-demand via search/feed APIs.

## 3. DATABASE — Missing Critical Indexes and Tables

**What's missing from your schema:**

```sql
-- User sessions (for analytics + mood inference)
CREATE TABLE user_sessions (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id),
    started_at TIMESTAMPTZ DEFAULT NOW(),
    ended_at TIMESTAMPTZ,
    device_type TEXT,
    app_version TEXT
);

-- Watch events (ground truth for ALL algorithms)
CREATE TABLE watch_events (
    id BIGSERIAL PRIMARY KEY,
    user_id INT NOT NULL,
    content_id INT NOT NULL,
    content_type TEXT NOT NULL, -- 'short' or 'challenge'
    watch_duration_ms INT NOT NULL,
    total_duration_ms INT NOT NULL,
    completed BOOLEAN DEFAULT FALSE,
    skipped BOOLEAN DEFAULT FALSE,
    rewatched BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);
